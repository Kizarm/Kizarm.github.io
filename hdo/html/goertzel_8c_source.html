18  * calc_coeffs

20  */

22 // #define NOCOMPUTE // ! pokud se toto definuje, zmenší se délka kódu podstatně, počítá vždy ve float
23 aritfmt calc_coeff (float nfreq) {
24 #ifdef NOCOMPUTE
25  return 1706; // napevno pro 216,6 Hz
26 #else // NOCOMPUTE - je to tu pro ukázku, jak se to počítá
27  aritfmt reval;
28  float coeff;
29  coeff = 2.0 * 3.141592654 * nfreq;
30  coeff = 2.0 * cos (coeff);
31 #ifndef COMPUTE_FLOAT
32  // aby se to dalo počítat v int, tak to tady zvětšíme
33  coeff *= (float)(1 << ISHIFT);
34 #endif // COMPUTE_FLOAT
35 
36  reval = (aritfmt) coeff;
37  // TRACE_INFO ("coeff = %d\r\n", reval);
38  return reval;
39 #endif // NOCOMPUTE
40 }
41 /*----------------------------------------------------------------------------
42  * goertzel
43  *----------------------------------------------------------------------------
44  */
45 int goertzel (aritfmt coeff, short *samples, int n) {
46  register aritfmt rv,q0,q1=0,q2=0; // prefix register je použit pro případnou optimalizaci
47  register int i;
48  //aritfmt min=0,max=0; // hledáme přetečení
49 #ifdef COMPUTE_FLOAT
50  aritfmt a,b,c;
51 #endif // COMPUTE_FLOAT
52  // pro všechny vzorky
54  for (i=0; i<n; i++) {
55  q0 = coeff * q1;
56  // pokud byl coeff zvětšen, je třeba to tu zase zmenšit. Bez FPU je to ale rychlejší než počítat ve float.
57 #ifndef COMPUTE_FLOAT
58  q0 >>= ISHIFT; // zmenšení pro int
59 #endif // COMPUTE_FLOAT
60  q0 += (aritfmt) samples[i] - q2; // vlastní výpočet
61 
62  q2 = q1; // posuv o vzorek
63  q1 = q0; // rekurze
64  }
65  // po n vzorcích vypočteme kvadrát výkonu signálu, počítá se to jen jednou za čas (120 ms)
66 #ifdef COMPUTE_FLOAT
67  a = (float) q1;
68  b = (float) q2;
69  c = (float) coeff;
70  a /= (float)(1 << HSHIFT); // zmenšíme, jinak jsou tam moc velká čísla a přeteče to
71  b /= (float)(1 << HSHIFT);
72  // výpočet kvadrátu výkonu signálu pro danou frekvenci
73  rv = (a * a) + (b * b) - (c * a * b);
74  TRACE_INFO ("Presna hodnota = %f\n", r);
75  return (int) r;
76 #else
77  // Předchozí jde udělat i v integer-32, ale není to tak přesné
78  q1 >>= HSHIFT;
79  q2 >>= HSHIFT;
80  rv = q1 * q2;
81  // TRACE_INFO ("q1*q2=%d\n", r);
82  rv *= -coeff;
83  rv >>= ISHIFT; // tady nutno zmenšit tak, jak bylo zvětšeno v calc_coeff()
84  rv += q1 * q1 + q2 * q2; // výkon by byl sqrt (rv), není nutné počítat, napětí stačí
85  return (int) rv;
86 #endif // COMPUTE_FLOAT
87 }
